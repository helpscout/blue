# AnimateGroup

This component is a light wrapper for `TransitionGroup` from the [react-transition-group](https://github.com/reactjs/react-transition-group) library.

<Preview>
  <Story name="default">
    <div>
      <p>Stagger fade in</p>
      <AnimateGroup stagger>
        <Animate sequence="fade">
          <div>Fade in</div>
        </Animate>
        <Animate sequence="fade left">
          <div>Fade in + left</div>
        </Animate>
        <Animate sequence="fade">
          <div>Fade in</div>
        </Animate>
        <Animate sequence="fade left">
          <div>Fade in + left</div>
        </Animate>
      </AnimateGroup>
    </div>
  </Story>
</Preview>
